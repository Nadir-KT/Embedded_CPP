#include <iostream>

using namespace std;

int main()
{
    string cars[5] = {"Volvo", "BMW", "Ford", "Mazda", "Tesla"};
for (int i = 0; i < 5; i++) 
{
  cout << i << " = " << cars[i] << "\n";
}
}

    2 = Ford
    3 = Mazda
